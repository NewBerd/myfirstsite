def sigmoid_derivative(x):#求sigmoid的导数:s(x)[1-s(x)]
    return tf.multiply(tf.sigmoid(x),tf.subtract(tf.constant(1.0),tf.sigmoid(x)))
#求relu的导数
def relu_derivative(x):
    cond=~tf.less(x,tf.zeros(tf.shape(x),dtype=x.dtype))#x不小于0为真,~表示取反
    out=tf.where(cond,tf.ones(tf.shape(x)),tf.zeros(tf.shape(x)))#当为真时，取1，为假（x<0）时取0
    return out
#求softmax的导数：s(x)[1-s(x)]
def softmax_derivative(x):#x必须为一个向量(或张量，但需指定axis=-1),并且不能为整型
    soft=tf.nn.softmax(x)
    d_sf=soft*(1-soft)
    return d_sf

def multilayers_perceptron(x,weights,biases):
    '''命名规则：每一层的输出为该层的名字（小写，例如隐层输出为h）；该层激活后的输出名字记为层名加下划线(例如，隐层激活后为h_)'''
    h=tf.matmul(x,weights['h1'])+biases['h1']#隐层线性单元输出
    h_=tf.nn.relu(h)#隐层激活后输出
#     h_=tf.sigmoid(h)
    
    o=tf.add(tf.matmul(h_,weights['out']),biases['out'])#输出层线性单元输出
#     o_=tf.nn.softmax(o)#softmax激活后输出
    o_=tf.sigmoid(o)
    return [o_,o,h_,h]

def multilayer_perceptron(x):
    fc1=layers.fully_connected(x,n_hidden,activation_fn=tf.nn.relu,scope='fc1_weights')#fc1是该层的权重变量
    out=layers.fully_connected(fc1,n_class,activation_fn=None,scope='out_weights')#out是输出层的权重变量
    return out
